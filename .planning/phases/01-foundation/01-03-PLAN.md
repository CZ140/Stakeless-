const parsed = envSchema.safeParse(process.env);

if (!parsed.success) {
  const issues = parsed.error.issues
    .map((issue) => `  - ${issue.path.join('.')}: ${issue.message}`)
    .join('\n');
  console.error(`[startup] Environment validation failed:\n${issues}`);
  console.error('[startup] Copy .env.example to .env and fill in the required values.');
  process.exit(1);
}

export const env = parsed.data;
export type Env = typeof env;

Create **apps/backend/src/app.ts** — Express factory function. Middleware order matters: security headers (helmet) first, then cors, then body parsing, then routes:
```typescript
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import { healthRouter } from './routes/health.js';

export function createApp() {
  const app = express();

  // Security headers first
  app.use(helmet());

  // CORS — allow requests from Vite dev server
  app.use(
    cors({
      origin: ['http://localhost:5173'],
      credentials: true,
    })
  );

  // Body parsing
  app.use(express.json());

  // Routes
  app.use('/api', healthRouter);

  return app;
}
```

Create **apps/backend/src/index.ts** — entry point. Import env FIRST (before any other app code) so the crash-on-missing-env fires before any other module loads:

import './env.js'; // Must be first — crashes with clear error if env vars are missing

Note: `.js` extensions on all relative imports are required for NodeNext moduleResolution (TypeScript compiles .ts → .js; Node imports the .js at runtime).

import react from '@vitejs/plugin-react';

export default defineConfig({
  plugins: [react()],
  server: {
    port: 5173,

      '/api': {
        target: 'http://localhost:3000',
