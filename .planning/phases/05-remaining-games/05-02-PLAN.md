    import { randomInt } from 'node:crypto';

    export interface MinesSessionState {
      mineCount: number;
      grid: boolean[];       // 25 elements, true = mine
      revealed: number[];    // safe tile indices revealed so far
      tilesRevealed: number;
      multiplier: number;
      status: 'active' | 'cashout' | 'exploded';
    }

    // Fisher-Yates shuffle using crypto.randomInt for fairness (GINF-02)
    export function generateMineGrid(mineCount: number): boolean[] {
      const grid = Array(25).fill(false);
      for (let i = 0; i < mineCount; i++) grid[i] = true;
      // Shuffle
      for (let i = 24; i > 0; i--) {
        const j = randomInt(0, i + 1);
        [grid[i], grid[j]] = [grid[j], grid[i]];
      }
      return grid;
    }

    // Multiplier after `tilesRevealed` safe reveals with `mineCount` mines in 25-tile grid.
    // Uses hypergeometric compound formula with 0.97 house factor.
    export function calculateMinesMultiplier(mineCount: number, tilesRevealed: number): number {
      if (tilesRevealed === 0) return 1.0;
      const totalSafe = 25 - mineCount;
      let multiplier = 1.0;
      for (let i = 0; i < tilesRevealed; i++) {
        multiplier *= (25 - i) / (totalSafe - i);
      }
      return Math.round(multiplier * 0.97 * 100) / 100; // house edge + 2dp rounding
    }

    **POST /api/games/mines/start**
    Schema: { betAmount: z.number().int().min(1), mineCount: z.number().int().min(1).max(24) }
    Pipeline:
    1. deductBet(userId, betAmount, 'mines')
    2. generateMineGrid(mineCount) → grid
    3. state: MinesSessionState = { mineCount, grid, revealed: [], tilesRevealed: 0, multiplier: 1.0, status: 'active' }
    4. db.insert(gameSessions).values({ userId, gameType: 'mines', state: JSON.stringify(state), betAmount }) → returns session id
    5. Return { sessionId: id } — grid NOT returned

    **POST /api/games/mines/tile**
    Schema: { sessionId: z.number().int(), row: z.number().int().min(0).max(4), col: z.number().int().min(0).max(4) }
    Pipeline:
    1. Load session: db.select().from(gameSessions).where(and(eq(gameSessions.id, sessionId), eq(gameSessions.userId, userId), isNull(gameSessions.completedAt)))
    2. If not found: 404 { error: 'Session not found' }
    3. Parse state = JSON.parse(session.state) as MinesSessionState
    4. If state.status !== 'active': 400 { error: 'Round already ended' }
    5. tileIndex = row * 5 + col
    6. If tileIndex in state.revealed: 400 { error: 'Tile already revealed' }
    7. If state.grid[tileIndex] === true (mine hit):
       - state.status = 'exploded'
       - db.update(gameSessions).set({ state: JSON.stringify(state), completedAt: new Date() }).where(eq(gameSessions.id, sessionId))
       - settleBet(userId, 0, session.betAmount, 'exploded', 'mines')  // profit = 0, bet lost
       - Return { hit: true, mineGrid: state.grid }
    8. Safe tile:
       - state.revealed.push(tileIndex)
       - state.tilesRevealed++
       - state.multiplier = calculateMinesMultiplier(state.mineCount, state.tilesRevealed)
       - db.update(gameSessions).set({ state: JSON.stringify(state), updatedAt: new Date() }).where(eq(gameSessions.id, sessionId))
       - Return { hit: false, gem: true, currentMultiplier: state.multiplier, tilesRevealed: state.tilesRevealed }

    **POST /api/games/mines/cashout**
    Schema: { sessionId: z.number().int() }
    Pipeline:
    1. Load and validate session (same as above)
    2. Parse state; if status !== 'active': 400 { error: 'Round already ended' }
    3. If tilesRevealed === 0: 400 { error: 'Reveal at least one tile before cashing out' }
    4. payout = Math.floor(session.betAmount * state.multiplier)
    5. profit = payout  // deductBet already removed stake; settleBet receives pure payout
    6. state.status = 'cashout'
    7. db.update(gameSessions).set({ state: JSON.stringify(state), completedAt: new Date() }).where(eq(gameSessions.id, sessionId))
    8. { newBalance } = await settleBet(userId, profit, session.betAmount, `cashout_${state.tilesRevealed}`, 'mines')
    9. Return { payout, newBalance, mineGrid: state.grid }

    **GET /api/games/mines/active-session**
    No body params — uses req.user!.id
    1. db.select().from(gameSessions).where(and(eq(gameSessions.userId, userId), eq(gameSessions.gameType, 'mines'), isNull(gameSessions.completedAt)))
    2. Take most recent if multiple (orderBy createdAt desc, limit 1)
    3. If none: return { session: null }
    4. Parse state; return { session: { sessionId: session.id, tilesRevealed: state.tilesRevealed, revealed: state.revealed, mineCount: state.mineCount, multiplier: state.multiplier, betAmount: session.betAmount } }
    — mine positions NOT included
